10) Write a functions for the mean, median, mode, range, and standard deviation

#Mean
Mean <- function(x){
  sum(x)/length(x)
}

x <- c(1,2,3,4,5,6)
Mean(x)
## [1] 3.5
#Median

#Mode
Mode <- function(x) {
 ux <- unique(x)
 ux[which.max(tabulate(match(x, ux)))]
}

x <- c(1,2,3,3,3)
Mode(x)
## [1] 3
#Range

Range <- function(x) {
  max(x) - min(x)
}

x <- c(1,2,3,3,3)
Range(x)
## [1] 2
#Standard Deviation

StandardDev <- function(x) {
  sqrt(sum((x- sum(x)/length(x))^2/(length(x)-1)))
}
